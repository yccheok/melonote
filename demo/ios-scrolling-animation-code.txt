
override func viewDidAppear(_ animated: Bool) {
        super.viewDidAppear(animated)

        // Step 1: wait 1 second
        DispatchQueue.main.asyncAfter(deadline: .now() + 1.0) { [weak self] in
            guard let self = self else { return }

            // Step 2: calculate the bottom offset
            let bottomOffset = CGPoint(
                x: 0,
                y: self.scrollView.contentSize.height - self.scrollView.bounds.height + 32
            )

            // Ensure there's something to scroll
            guard bottomOffset.y > 0 else { return }
            
            // Step 3: animate scrolling over 10 seconds
            UIView.animate(withDuration: 20.0, delay: 0, options: .curveLinear) {
                self.scrollView.setContentOffset(bottomOffset, animated: false)
            }
        }
}
